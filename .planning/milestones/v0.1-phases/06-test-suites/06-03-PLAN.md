```typescript
import { describe, it, expect, beforeEach, afterEach, vi } from 'vitest';
import WS from 'vitest-websocket-mock';
import type { ReactiveControllerHost } from 'lit';
import { ChatStore } from './chat-store.js';

function createMockHost(): ReactiveControllerHost {
  return {
    addController: vi.fn(),
    removeController: vi.fn(),
    requestUpdate: vi.fn(),
    updateComplete: Promise.resolve(true),
  };
}

const URL = 'ws://localhost:1234';
let server: WS;
let store: ChatStore;

beforeEach(() => {
  server = new WS(URL);
  store = new ChatStore(createMockHost());
});

afterEach(() => {
  store.disconnect();
  WS.clean();
});
```

**TEST-04 -- Full message flow** (`describe('full message flow')`):

```
it('connect -> send -> stream tokens -> finalize', async () => {
```
1. `store.connect(URL, false)` -- assert connectionState='connecting'
2. `await server.connected` -- WebSocket connected
3. `server.send(JSON.stringify({ type: 'connected', session_id: 'test-123' }))` -- use `vi.waitFor` or microtask flush to assert connectionState='connected', inputDisabled=false
4. `store.send('Hello')` -- assert messages has 1 user message with content 'Hello'
5. `await expect(server).toReceiveMessage(JSON.stringify({ type: 'message', content: 'Hello' }))` -- verify server received the message
6. `server.send(JSON.stringify({ type: 'token', content: 'Hi' }))` -- flush microtasks, assert messages length=2, messages[1].streaming=true, messages[1].content='Hi'
7. `server.send(JSON.stringify({ type: 'token', content: ' there' }))` -- assert messages[1].content='Hi there'
8. `server.send(JSON.stringify({ type: 'message_end' }))` -- assert messages[1].streaming=false, messages[1].content='Hi there'

**Timing note:** After each `server.send()`, ChatClient processes the WebSocket message synchronously via onmessage, dispatches a CustomEvent, and ChatStore's handler updates state. This should be synchronous within the event loop, but if assertions fail, wrap them in `vi.waitFor(() => { expect(...) })`.
